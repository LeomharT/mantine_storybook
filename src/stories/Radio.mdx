### Controlled

```js
import { useState } from 'react';
import { Radio } from '@mantine/core';

function Demo() {
	const [checked, setChecked] = useState(false);
	return (
		<Radio
			checked={checked}
			onChange={(event) => setChecked(event.currentTarget.checked)}
		/>
	);
}
```
